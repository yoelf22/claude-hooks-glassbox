# glassBox Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Claude Code hook plugin that explains every tool call in plain English, with caching and a "learn" mechanism to dismiss familiar tools.

**Architecture:** A single bash script (`glassbox.sh`) registered as a `PreToolUse` hook. It reads tool call JSON from stdin, normalizes it, checks a file-based cache, and on miss calls `claude -p` to generate an explanation. A companion CLI (`glassbox`) manages learn/unlearn/cache operations.

**Tech Stack:** Bash, jq, claude CLI, shasum

# Normalize input for cache key generation
# Bash: extract command name + flags, strip positional args
# Other tools: extract parameter key names only
normalize() {
    local tool="$1"
    local input="$2"

    if [ "$tool" = "Bash" ]; then
        local cmd
        cmd=$(echo "$input" | jq -r '.command // empty' 2>/dev/null)
        [ -z "$cmd" ] && echo "Bash:" && return

        # For compound commands (&&, ||, ;), process the full pipeline
        # Extract program names and flags, strip positional arguments
        local normalized=""
        local in_compound=false

        # Split on && and process each part
        echo "$cmd" | tr '&' '\n' | grep -v '^$' | while IFS= read -r part; do
            part=$(echo "$part" | sed 's/^[[:space:]]*//;s/[[:space:]]*$//')
            [ -z "$part" ] && continue

            local result=""
            local first=true
            for word in $part; do
                if $first; then
                    # First word is the command
                    result="$word"
                    first=false
                elif [[ "$word" == -* ]]; then
                    # Flags (start with -)
                    result="$result $word"
                fi
                # Skip positional args (paths, URLs, names, values)
            done
            if [ -n "$normalized" ]; then
                normalized="$normalized && $result"
            else
                normalized="$result"
            fi
        done

        # Fallback: simpler approach — process word by word
        local result=""
        local first=true
        for word in $cmd; do
            if [ "$word" = "&&" ] || [ "$word" = "||" ] || [ "$word" = ";" ]; then
                result="$result $word"
                first=true
            elif $first; then
                [ -n "$result" ] && result="$result "
                result="${result}${word}"
                first=false
            elif [[ "$word" == -* ]]; then
                result="$result $word"
            fi
        done
        echo "Bash:$result"
    else
        # Non-Bash: sorted list of parameter keys
        local keys
        keys=$(echo "$input" | jq -r '.tool_input // . | keys | sort | join(",")' 2>/dev/null)
        echo "${tool}:${keys}"
    fi
}

# --normalize-only mode for testing
if [ "${1:-}" = "--normalize-only" ]; then
    normalize "$TOOL_NAME" "$TOOL_INPUT"
    exit 0
fi

# Generate a 16-char hex cache key from a normalized string
cache_key() {
    echo -n "$1" | shasum -a 256 | cut -c1-16
}

# Look up a cached explanation. Returns empty string on miss.
cache_lookup() {
    local key="$1"
    local file="${CACHE_DIR}/${key}"
    if [ -f "$file" ]; then
        cat "$file"
    fi
}

# Store an explanation in the cache.
cache_store() {
    local key="$1"
    local content="$2"
    mkdir -p "$CACHE_DIR"
    echo "$content" > "${CACHE_DIR}/${key}"
}

# Purge cache entries older than 30 days.
cache_cleanup() {
    [ -d "$CACHE_DIR" ] && find "$CACHE_DIR" -type f -mtime +30 -delete 2>/dev/null
}
```

# Check if a tool (or tool:command pattern) is marked as learned
is_learned() {
    local pattern="$1"
    [ -f "$LEARNED_FILE" ] || return 1
    # Check exact match first
    grep -qFx "$pattern" "$LEARNED_FILE" 2>/dev/null && return 0
    # For Bash:cmd patterns, check if the base tool is learned
    local base="${pattern%%:*}"
    grep -qFx "$base" "$LEARNED_FILE" 2>/dev/null && return 0
    # For Bash:git log, check if Bash:git is learned
    if [ "$base" = "Bash" ]; then
        local cmd="${pattern#Bash:}"
        local first_word="${cmd%% *}"
        grep -qFx "Bash:${first_word}" "$LEARNED_FILE" 2>/dev/null && return 0
    fi
    return 1
}

# Mark a tool as learned
learn_tool() {
    mkdir -p "$GLASSBOX_DIR"
    echo "$1" >> "$LEARNED_FILE"
}

# Remove a tool from learned list
unlearn_tool() {
    [ -f "$LEARNED_FILE" ] || return 0
    local tmp="${LEARNED_FILE}.tmp"
    grep -vFx "$1" "$LEARNED_FILE" > "$tmp" 2>/dev/null || true
    mv "$tmp" "$LEARNED_FILE"
}
```

#!/bin/bash
# glassbox — CLI for managing glassBox settings
GLASSBOX_DIR="${HOME}/.glassbox"
LEARNED_FILE="${GLASSBOX_DIR}/learned"
CACHE_DIR="${GLASSBOX_DIR}/cache"

case "${1:-help}" in
    learn)
        [ -z "${2:-}" ] && echo "Usage: glassbox learn <tool>" && exit 1
        mkdir -p "$GLASSBOX_DIR"
        echo "$2" >> "$LEARNED_FILE"
        echo "Learned: $2 — will show brief only."
        ;;
    unlearn)
        [ -z "${2:-}" ] && echo "Usage: glassbox unlearn <tool>" && exit 1
        if [ -f "$LEARNED_FILE" ]; then
            tmp="${LEARNED_FILE}.tmp"
            grep -vFx "$2" "$LEARNED_FILE" > "$tmp" 2>/dev/null || true
            mv "$tmp" "$LEARNED_FILE"
            echo "Unlearned: $2 — will show full explanations again."
        else
            echo "Nothing learned yet."
        fi
        ;;
    learned)
        if [ -f "$LEARNED_FILE" ] && [ -s "$LEARNED_FILE" ]; then
            echo "Learned tools (brief only):"
            while IFS= read -r line; do
                echo "  - $line"
            done < "$LEARNED_FILE"
        else
            echo "No tools learned yet. Use 'glassbox learn <tool>' to dismiss."
        fi
        ;;
    clear-cache)
        if [ -d "$CACHE_DIR" ]; then
            count=$(find "$CACHE_DIR" -type f | wc -l | tr -d ' ')
            rm -rf "$CACHE_DIR"
            echo "Cleared $count cached explanations."
        else
            echo "Cache is empty."
        fi
        ;;
    help|*)
        echo "glassBox — transparency for Claude Code"
        echo ""
        echo "Commands:"
        echo "  learn <tool>     Dismiss a tool (show brief only)"
        echo "  unlearn <tool>   Re-enable full explanations"
        echo "  learned          List dismissed tools"
        echo "  clear-cache      Clear explanation cache"
        echo ""
        echo "Tools: Read, Write, Edit, Grep, Glob, Bash, Agent"
        echo "Bash commands: Bash:git, Bash:npm, Bash:python, etc."
        ;;
esac

```bash
# Build the LLM prompt for generating an explanation
build_prompt() {
    local tool="$1"
    local normalized_input="$2"
    local prompt="You are glassBox, explaining Claude Code tools to a non-technical user.

Tool: ${tool}
Input: ${normalized_input}

Respond with exactly two sections:
BRIEF: A one-liner (max 10 words) of what this does.
EXTENDED: A plain-English explanation of the command/tool and its flags/arguments. Do not include specific file paths, URLs, branch names, or user values. Explain what the arguments DO, not what they ARE. Keep it under 60 words."

    # For Bash commands, try to enrich with man/help output
    if [ "$tool" = "Bash" ]; then
        local base_cmd="${normalized_input%% *}"
        local ref=""
        ref=$(man "$base_cmd" 2>/dev/null | head -30) || \
        ref=$("$base_cmd" --help 2>&1 | head -20) || true
        if [ -n "$ref" ]; then
            prompt="${prompt}

Reference (from man/--help):
${ref}"
        fi
    fi

    echo "$prompt"
}

# Parse the LLM response into brief and extended parts
parse_response() {
    local response="$1"
    local part="$2"

    if [ "$part" = "brief" ]; then
        echo "$response" | grep '^BRIEF:' | sed 's/^BRIEF:[[:space:]]*//'
    elif [ "$part" = "extended" ]; then
        echo "$response" | sed -n '/^EXTENDED:/,$ p' | sed 's/^EXTENDED:[[:space:]]*//'
    fi
}

# Call Claude to generate an explanation (returns raw response)
generate_explanation() {
    local prompt="$1"
    claude -p "$prompt" 2>/dev/null || echo ""
}

# === Main execution ===

# Ensure cache dir exists
mkdir -p "$CACHE_DIR" 2>/dev/null || true

NORMALIZED=$(normalize "$TOOL_NAME" "$TOOL_INPUT")

if [ -n "$CACHED" ]; then
    BRIEF=$(echo "$CACHED" | head -1)
    EXTENDED=$(echo "$CACHED" | tail -n +2)
else
    # Generate explanation via LLM
    PROMPT=$(build_prompt "$TOOL_NAME" "${NORMALIZED#*:}")
    RESPONSE=$(generate_explanation "$PROMPT")

    if [ -n "$RESPONSE" ]; then
        BRIEF=$(parse_response "$RESPONSE" "brief")
        EXTENDED=$(parse_response "$RESPONSE" "extended")

        # Cache the result
        if [ -n "$BRIEF" ]; then
            cache_store "$KEY" "${BRIEF}
${EXTENDED}"
        fi
    fi
fi

# Output to stderr
if [ -n "$BRIEF" ]; then
    # Header: icon + tool name + brief
    echo "  ◆ ${TOOL_NAME}  ${BRIEF}" >&2

    # Extended explanation (skip if tool is learned)
    if [ -n "$EXTENDED" ] && ! is_learned "$NORMALIZED"; then
        echo "$EXTENDED" | fmt -w 72 | while IFS= read -r line; do
            echo "    $line" >&2
        done
    fi
fi

exit 0

    mkdir -p "$(dirname "$SETTINGS_FILE")"
    echo '{}' > "$SETTINGS_FILE"
fi

# Check if hook already registered
if jq -e '.hooks.PreToolUse[]? | select(.hooks[]?.command == "'"$GLASSBOX_SCRIPT"'")' "$SETTINGS_FILE" &>/dev/null; then
    echo "  Hook already registered."
else
    # Add the PreToolUse hook
    UPDATED=$(jq --arg cmd "$GLASSBOX_SCRIPT" '
        .hooks.PreToolUse = (.hooks.PreToolUse // []) + [{
            "matcher": "*",
            "hooks": [{
                "type": "command",
                "command": $cmd
            }]
        }]
    ' "$SETTINGS_FILE")
    echo "$UPDATED" > "$SETTINGS_FILE"
    echo "  Registered PreToolUse hook in ${SETTINGS_FILE}"
fi

echo ""
echo "glassBox installed. Restart Claude Code to activate."
```

# glassBox

A Claude Code hook that explains what every tool does while Claude works — in plain English.

When Claude uses a tool (Bash, Read, Edit, Grep, etc.), glassBox prints a brief description followed by a plain-English explanation of what the tool does and what its arguments mean. Explanations are generated by Claude and cached locally.

## Install

```bash
git clone <repo-url> && cd glassbox
./install.sh
```

Requires: `jq`, `claude` CLI, `shasum`

## How it works

glassBox registers a `PreToolUse` hook that fires before every tool call. It:

1. Reads the tool name and input
2. Normalizes the input (strips file paths, URLs, secrets — only structure remains)
3. Checks a local cache (`~/.glassbox/cache/`)
4. On miss, calls Claude to generate an explanation
5. Prints the explanation to stderr

Your data stays private — no file paths, URLs, branch names, or secrets are sent to the LLM.

## Learn / dismiss tools

Once you understand what a tool does, dismiss it:

```bash
glassbox learn Read        # dismiss the Read tool
glassbox learn Bash:git    # dismiss all git commands
glassbox learned           # list dismissed tools
glassbox unlearn Read      # re-enable full explanations
```

Learned tools still show the brief one-liner, just not the extended explanation.

## Uninstall

```bash
./uninstall.sh
```

## Cache

Explanations are cached in `~/.glassbox/cache/` and auto-expire after 30 days. Clear manually with:

```bash
glassbox clear-cache
```
```

#!/bin/bash
# tests/test_integration.sh — end-to-end test of the full hook flow

assert_eq() {
    local desc="$1" expected="$2" actual="$3"
    if [ "$expected" = "$actual" ]; then
        echo "  PASS: $desc"
        ((PASS++))
    else
        echo "  FAIL: $desc"
        echo "    expected: $expected"
        echo "    actual:   $actual"
        ((FAIL++))
    fi
}

# Test: script exits 0 on valid input (even without claude CLI)
input='{"tool_name":"Grep","tool_input":{"pattern":"test","type":"js"}}'

assert_eq "exits 0 on valid input" "0" "$?"

# Test: script exits 0 on malformed input
echo "garbage" | bash "$SCRIPT_DIR/glassbox.sh" 2>/dev/null
assert_eq "exits 0 on garbage" "0" "$?"

# Test: script exits 0 on empty input
echo "" | bash "$SCRIPT_DIR/glassbox.sh" 2>/dev/null
assert_eq "exits 0 on empty" "0" "$?"

assert_eq "cached output shown" "true" "$(echo "$output" | grep -q 'Reading a file' && echo true || echo false)"

assert_eq "learned: brief shown" "true" "$has_brief"
assert_eq "learned: extended hidden" "false" "$has_extended"

rm -rf "$GLASSBOX_DIR"

echo ""
echo "Results: $PASS passed, $FAIL failed"
[ "$FAIL" -eq 0 ] && exit 0 || exit 1
